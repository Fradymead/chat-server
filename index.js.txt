const WebSocket = require("ws");
const PORT = process.env.PORT || 10000;

const wss = new WebSocket.Server({ port: PORT }, () => {
  console.log(`âœ… Server WebSocket attivo su ws://localhost:${PORT}`);
});

wss.on("connection", (ws) => {
  console.log("ğŸ”Œ Nuova connessione");

  ws.on("message", (message) => {
    try {
      const data = JSON.parse(message);
      console.log("ğŸ“¨ Messaggio ricevuto:", data);

      // Inoltra il messaggio a tutti i client
      wss.clients.forEach((client) => {
        if (client !== ws && client.readyState === WebSocket.OPEN) {
          client.send(JSON.stringify(data));
        }
      });
    } catch (err) {
      console.error("âŒ Messaggio non valido:", message);
    }
  });

  ws.on("close", () => console.log("â Connessione chiusa"));
});
